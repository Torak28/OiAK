beta = 4
for n in range(1, 100):
    for k in range(1, 10000):
        rns = [(pow(2,n))-k,(pow(2,n))+k,(pow(2,n))+1,(pow(2,n))-1,(pow(2,n+beta))]
        if(rns[0] != 0):
            v1 = rns[1] % rns[0]
            v2 = rns[2] % rns[0]
            v3 = rns[3] % rns[0]
            v4 = rns[4] % rns[0]

            v = k * v1 * v2 * v3 * v4
            wynik = v % rns[0]
            if(wynik == 1):
